public class TestMath {

    private Integer num1;
    private Integer num2;

    public TestMath() {
        this.num1 = 0;
        this.num2 = 0;
    }

    public TestMath(Integer n1, Integer n2) {
        this.num1 = n1;
        this.num2 = n2;
    }

    public void setNum1(Integer n) {
        this.num1 = n;
    }

    public void setNum2(Integer n) {
        this.num2 = n;
    }

    public Integer plus() {
        return num1 + num2;
    }

    public Integer minus() {
        return num1 - num2;
    }

}

# javac TestMath.java

# touch MANIFEST.MF

# jar -cvfm TestMath.jar MANIFEST.MF ./

佈建php/java bridge

jar放置到tomcat/lib中

# mv TestMath.jar tomcat/lib/

將上面下載下來的war丟掉tomcat/webapps目錄中

啟動tomcat即完成
